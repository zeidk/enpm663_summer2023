313 bool FloorRobot::move_through_waypoints_(
314  std::vector<geometry_msgs::msg::Pose> waypoints, double vsf, double asf)
315 {
316  moveit_msgs::msg::RobotTrajectory trajectory;
317 
318  double path_fraction = floor_robot_.computeCartesianPath(waypoints, 0.01, 0.0, trajectory);
319 
320  if (path_fraction < 0.9)
321  {
322  RCLCPP_ERROR(get_logger(), "Unable to generate trajectory through waypoints");
323  return false;
324  }
325 
326  // Retime trajectory
327  robot_trajectory::RobotTrajectory rt(floor_robot_.getCurrentState()->getRobotModel(), "floor_robot");
328  rt.setRobotTrajectoryMsg(*floor_robot_.getCurrentState(), trajectory);
329  totg_.computeTimeStamps(rt, vsf, asf);
330  rt.getRobotTrajectoryMsg(trajectory);
331 
332  return static_cast<bool>(floor_robot_.execute(trajectory));
333 }

336 void FloorRobot::wait_for_attach_completion_(double timeout)
337 {
338  // Wait for part to be attached
339  rclcpp::Time start = now();
340  std::vector<geometry_msgs::msg::Pose> waypoints;
341  geometry_msgs::msg::Pose starting_pose = floor_robot_.getCurrentPose().pose;
342 
343  while (!floor_gripper_state_.attached)
344  {
345  RCLCPP_INFO_THROTTLE(get_logger(), *get_clock(), 1000, "Waiting for gripper attach");
346 
347  waypoints.clear();
348  starting_pose.position.z -= 0.001;
349  waypoints.push_back(starting_pose);
350 
351  move_through_waypoints_(waypoints, 0.1, 0.1);
352 
353  usleep(200);
354 
355  if (now() - start > rclcpp::Duration::from_seconds(timeout))
356  {
357  RCLCPP_ERROR(get_logger(), "Unable to pick up object");
358  return;
359  }
360  }
361 }

583  {
584  if (part.part.type == part_to_pick.type && part.part.color == part_to_pick.color)
585  {
586  part_pose = Utils::multiply_poses(right_bins_camera_pose_, part.pose);
587  found_part = true;
588  bin_side = "right_bins";
589  break;
590  }
591  }
592  }
593  if (!found_part)
594  {
595  RCLCPP_ERROR(get_logger(), "Unable to locate part");
596  return false;
597  }
598 
599  double part_rotation = Utils::get_yaw_from_pose_(part_pose);
600 
601  // Change gripper at location closest to part
602  if (floor_gripper_state_.type != "part_gripper")
603  {
604  std::string station;
605  if (part_pose.position.y < 0)
606  {
607  station = "kts1";
608  }
609  else
610  {
611  station = "kts2";
612  }
613 
614  // Move floor robot to the corresponding kit tray table
615  if (station == "kts1")
616  {
617  floor_robot_.setJointValueTarget(floor_kts1_js_);
618  }
619  else
620  {
621  floor_robot_.setJointValueTarget(floor_kts2_js_);
622  }
623  move_to_target_();
624 
625  change_gripper_(station, "parts");
626  }
627 
628  floor_robot_.setJointValueTarget("linear_actuator_joint", rail_positions_[bin_side]);
629  floor_robot_.setJointValueTarget("floor_shoulder_pan_joint", 0);
630  move_to_target_();
631 
632  std::vector<geometry_msgs::msg::Pose> waypoints;
633  waypoints.push_back(Utils::build_pose(part_pose.position.x, part_pose.position.y,
634  part_pose.position.z + 0.5, set_robot_orientation_(part_rotation)));
635 
636  waypoints.push_back(Utils::build_pose(part_pose.position.x, part_pose.position.y,
637  part_pose.position.z + part_heights_[part_to_pick.type] + pick_offset_, set_robot_orientation_(part_rotation)));
638 
639  move_through_waypoints_(waypoints, 0.3, 0.3);
640 
641  set_gripper_state_(true);
642 
643  wait_for_attach_completion_(3.0);
644 
645  // Add part to planning scene
646  std::string part_name = part_colors_[part_to_pick.color] + "_" + part_types_[part_to_pick.type];
647  add_single_model_to_planning_scene_(part_name, part_types_[part_to_pick.type] + ".stl", part_pose);
648  floor_robot_.attachObject(part_name);
649  floor_robot_attached_part_ = part_to_pick;
650 
651  // Move up slightly
652  waypoints.clear();
653  waypoints.push_back(Utils::build_pose(part_pose.position.x, part_pose.position.y,
654  part_pose.position.z + 0.3, set_robot_orientation_(0)));
655 
656  move_through_waypoints_(waypoints, 0.3, 0.3);
657 
658  return true;
659 }

662 bool FloorRobot::place_part_in_tray_(int agv_num, int quadrant)
663 {
664  if (!floor_gripper_state_.attached)
665  {
666  RCLCPP_ERROR(get_logger(), "No part attached");
667  return false;
668  }
669 
670  // Move to agv
671  floor_robot_.setJointValueTarget("linear_actuator_joint", rail_positions_["agv" + std::to_string(agv_num)]);
672  floor_robot_.setJointValueTarget("floor_shoulder_pan_joint", 0);
673  move_to_target_();
674 
675  // Determine target pose for part based on agv_tray pose
676  auto agv_tray_pose = get_pose_in_world_frame_("agv" + std::to_string(agv_num) + "_tray");
677 
678  auto part_drop_offset = Utils::build_pose(quad_offsets_[quadrant].first, quad_offsets_[quadrant].second, 0.0,
679  geometry_msgs::msg::Quaternion());
680 
681  auto part_drop_pose = Utils::multiply_poses(agv_tray_pose, part_drop_offset);
682 
683  std::vector<geometry_msgs::msg::Pose> waypoints;
684 
685  waypoints.push_back(Utils::build_pose(part_drop_pose.position.x, part_drop_pose.position.y,
686  part_drop_pose.position.z + 0.3, set_robot_orientation_(0)));
687 
688  waypoints.push_back(Utils::build_pose(part_drop_pose.position.x, part_drop_pose.position.y,
689  part_drop_pose.position.z + part_heights_[floor_robot_attached_part_.type] + drop_height_,
690  set_robot_orientation_(0)));
691 
692  move_through_waypoints_(waypoints, 0.3, 0.3);
693 
694  // Drop part in quadrant
695  set_gripper_state_(false);
696 
697  std::string part_name = part_colors_[floor_robot_attached_part_.color] +
698  "_" + part_types_[floor_robot_attached_part_.type];
699  floor_robot_.detachObject(part_name);
700 
701  waypoints.clear();
702  waypoints.push_back(Utils::build_pose(part_drop_pose.position.x, part_drop_pose.position.y,
703  part_drop_pose.position.z + 0.3,
704  set_robot_orientation_(0)));
705 
706  move_through_waypoints_(waypoints, 0.2, 0.1);
707 
708  return true;
709 }
